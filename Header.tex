\newcommand{\tab}{\hspace*{0.5cm}}

\AtBeginSubsection[]  % "Beamer, do the following at the start of every subsection"
{
\begin{frame}[c]{Outline}
\tableofcontents[ 
sectionstyle=show/hide, % current_section, other_section
subsectionstyle=show/shaded/hide % current_subsection, other_subsection, other_subsection_of _other_section
] 
\end{frame}
}

\AtBeginSection[]  % "Beamer, do the following at the start of every section"
{
\begin{frame}[c]{Outline}
\begin{multicols*}{2}
	\tableofcontents[ 
		sectionstyle=show/shaded, 
		subsectionstyle=hide/hide/hide
	]
\end{multicols*}	
\end{frame}
}
